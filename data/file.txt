عن هذا الدليل
ّ يهـدف هـذا الدليـل إلـى زيـادة ظهـور مـزودي خدمـات إعـادة التدويـر فـي المراحـل المختلفـة
عبــر سالســل القيمــة فــي الســوق المحلــي، وتعزيــز التشــبيك والوصــول إلــى المعلومــات
والخدمــات بيــن مختلــف الجهــات الفاعلــة وخاصــة متلقــي الخدمــات ضمــن ســوق إعــادة
ّ التدويــر فــي المملكــة. هــذا األمــر يمكــن مــن تســهيل إنشــاء الروابــط واألعمــال التجاريــة
بيـن مشـتري وبائعـي المـواد القابلـة للتدويـر وخدمـات إعـادة التدويـر عبـر سالسـل القيمـة،
ً علــى احتياجــات متلقــي الخدمــة
إضافــة إلــى تســهيل الوصــول إلــى معلومــات الســوق بنــاء
ّ مــن القطاعــات التجاريــة فــي عمــان، بحيــث يوفــر هــذا الدليــل للمســتخدم معلومــات عــن
مـزودي الخدمـات والجهـات الفاعلـة فـي سـوق إعـادة التدويـر فـي المملكـة يمكـن الوصـول
إليهـا مـن أجـل تنميـة األعمـال وتعزيـز القيمـة المضافـة فـي مختلـف المجـاالت والمسـتويات
المتعلقــة بإعــادة التدويــر.
ويتضمـن الدليـل معلومـات عـن أكثـر مـن 100 ّ شـركة ومؤسسـة فرديـة من مـزودي الخدمات
ّ فـي السـوق المحلـي باإلضافـة إلـى أفـراد غيـر رسـمين مـن الفئـات المجتمعيـة المهمشـة
والذيــن يعملــون فــي مختلــف خدمــات إعــادة التدويــر، مــع تســليط الضــوء علــى فئــات
السـيدات والشـباب، خاصـة بـأن هـذا القطـاع يعـد متاحـاً لتنميـة وتطويـر األفـكار اإلبداعيـة
ومشـاركة الفئـات المجتمعيـة األكثـر تأثـراً مـن النسـاء والشـباب فـي مجـاالت مختلفـة تنـدرج
تحـت قطـاع إعـادة التدويـر.
ّ ويسـتعرض الدليـل اسـم المنشـأة و/أو مقـدم الخدمـة، والشـخص المسـوؤل، ومعلومـات
ّ االتصـال ونـوع الخدمـات المقدمـة، ونطـاق تغطيـة الخدمـة. والسـتدامة هـذا الدليـل، سـيتم
ّ تحديــث المعلومــات المتضمنــة عــن مــزودي خدمــات إعــادة التدويــر فــي ظــل التطــورات
والمسـتجدات المتعلقـة بنمـو سـوق إعـادة التدويـر فـي المملكـة.
”إن قوائــم الخدمــات والشــركات والمؤسســات واألفــراد المندرجــة فــي هــذه النســخة مــن

الدليـل تـم إعدادهـا بنـاء
واألفراد ذاتهم وال تشـمل بالضرورة جميع الشـركات والعاملين في هذا القطاع. والسـتدامة
ّ هــذا الدليــل، ســيتم تحديــث المعلومــات المتضمنــة عــن مــزودي خدمــات إعــادة التدويــر

مقدمة
يعـد قطـاع النفايـات البلديـة الصلبـة فـي األردن أحـد أكثـر القطاعـات تعقيـداً وذلـك بسـبب
تنــوع واختــاف النفايــات الصلبــة المتولــدة وتنــوع توزيــع المســؤوليات بيــن مختلــف
القطاعــات الحكوميــة والخاصــة العاملــة فــي هــذا القطــاع.
ومـن الجديـر بالذكـر أن مدينـة عمـان والتـي تعتبـر المدينـة األكبـر فـي األردن مـن حيـث عـدد
السـكان ومعـدل التحضـر تنتـج حاليـاً حوالـي 1.2 مليـون طـن مـن النفايـات البلديـة الصلبـة
سـنوياً بمعـدل زيـادة سـنوية مقدارهـا ،%5 حيـث تضـم هـذه النفايـات خليطـاً مـن النفايـات
المنزليــة والتجاريــة، مثــل نفايــات الطعــام، والبالســتيك، والمعــادن، والــورق والكرتــون،
والمنسـوجات، وزيـت الطهـي، واإلطـارات، واإللكترونيـات، والزجـاج، باإلضافـة إلـى المـواد
الصلبــة الطبيــة والصناعيــة الخطــرة وغيــر الخطــرة. وفقــاً لإلســتراتيجية الوطنيــة األردنيــة
إلدارة النفايـات الصلبـة، تتـراوح كميـة النفايـات الصلبـة المحليـة المنتجـة للفـرد مـن 0.87
كجـم للفـرد يوميـاً فـي المناطـق الريفيـة إلـى 0.99 كجـم للفـرد يوميـاً فـي المناطـق الحضرية.
هـذا ويتـم التخلـص مـن معظـم النفايـات البلديـة الصلبـة الناتجـة فـي مكـب الغبـاوي. بينمـا
يتـم اسـتعادة 5 - 10 % فقـط مـن النفايـات القابلـة إلعـادة التدويـر بشـكل غيـر رسـمي فـي
األردن.
مكـب الغبـاوي هـو المكـب الوحيـد للتخلـص مـن النفايـات الصلبة فـي عمان، حيث يسـتقبل
،ً وتســبب الزيــادة المضطــردة لكميــة
هــذا المكــب حوالــي 3200 طــن مــن النفايــات يوميــا
ً تقليـل للعمـر التشـغيلي للمكـب. وتقـدر أمانـة عمـان الكبـرى أن
النفايـات المتولـدة يوميـا
مـا ال يقـل عـن %40-30 مـن هـذه النفايـات ينتجهـا القطـاع التجـاري الـذي يضـم حوالـي 99
ألـف منشـأة تجاريـة مسـجلة فـي عمـان.

الحديد والمعادن
يعتبــر قطــاع الحديــد والمعــادن مــن القطاعــات المتطــورة فــي األردن ومنــذ فتــرة طويلــة،
كمـا أنـه قطـاع كبيـر الحجـم حيـث يعمـل علـى إعـادة تدويـر المعـادن الحديديـة بالكامـل وإعـادة
إســتخدامها محليــا،ً وتنتــج مصانــع الصهــر المحليــة حديــد التســليح والصفائــح المعدنيــة
وغيرهـا مـن المنتجـات المسـتخدمة بشـكل رئيسـي كمدخـات لقطـاع اإلنشـاءات المحلـي.
كمـا يتـم تصديـر المعـادن غيـر الحديديـة خاصـة األلمنيـوم والنحـاس اللـذان تتـم معالجتهمـا
بشــكل محــدود ممــا يحــد مــن القيمــة المضافــة محليــا.ً قــدرت الصــادرات األردنيــة مــن
المعـادن غيـر الحديديـة بمـا يتـراوح بيـن 15 و 25 ألـف طـن فـي عـام ،2019 وتشـير التقديـرات
ّ إلــى أن قطــاع المعــادن يوظــف حوالــي ســتة آالف عامــل رســمي بــدوام كامــل. وبإعتبارهــا
سـلعة عالميـة، تتأثـر أسـعار المعـادن بالمتغيـرات العالميـة ممـا يؤثـر علـى تنافسـية القطـاع
وتذبـذب األسـعار بشـكل عـام.
